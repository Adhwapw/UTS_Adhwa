Meeting 4: Pengenalan PHP

1. Contoh Variabel dalam PHP

<?php
$username = "Cecep";
$nama = 22;

echo "Username: " . $username . "<br>";
echo "Nama: " . $nama;
?>

2. Contoh Konstanta dalam PHP

<?php
define("SITE_NAME", "Pemrograman Web");
echo "Selamat datang di " . SITE_NAME;
?>

3. Contoh Operator Aritmatika

<?php
$a = 10;
$b = 5;
$hasil = $a + $b;
echo "Hasil penjumlahan: " . $hasil;
?>

4. Contoh Operator Relasional

<?php
$a = 10;
$b = 5;
$hasil = ($a > $b);
var_dump($hasil);
?>

5. Contoh Operator Logika

<?php
$a = true;
$b = false;
$hasil = $a && $b;
var_dump($hasil);
?>

6. Contoh Program Array

<?php
$nama = array("Cecep", "Asep", "Dudu");
echo "Nama pertama: " . $nama[0];
?>

7. Contoh Program OOP di PHP

<?php
class Mobil {
    public $merk;
    public $tahun;

    function __construct($merk, $tahun) {
        $this->merk = $merk;
        $this->tahun = $tahun;
    }

    function getInfo() {
        return "Mobil: " . $this->merk . " Tahun: " . $this->tahun;
    }
}

$myCar = new Mobil("Toyota", 2022);
echo $myCar->getInfo();
?>
