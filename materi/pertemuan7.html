Materi 7: Array

1. Contoh Program Array Berindeks

<?php
//Mendefinisikan array berindeks
$array = array(10, 20, 30, 40, 50);

//Mengakses elemen berdasarkan indeks
echo "Elemen pertama: ". $array[0] ."<br>"; // Output: 10
echo "Elemen kedua: ". $array[1] ."<br>"; // Output: 20
echo "Elemen ketiga: ". $array[2] ."<br>"; // Output: 30
echo "Elemen keempat: ". $array[3] ."<br>"; // Output: 40
echo "Elemen kelima: ". $array[5] ."<br>"; // Output: 50

?>

2. Contoh Program Array Asosiatif

<?php
$mahasiswa = array(
    "nama" => "Cecep",
    "umur" => 20,
    "jurusan" => "Informatika",
    "ipk" => 3.85
);

// Menampilkan nilai dari array asosiatif berdasarkan kunci
echo "Nama: " . $mahasiswa["nama"] . "\n";
echo "Umur: " . $mahasiswa["umur"] . "\n";
echo "Jurusan: " . $mahasiswa["jurusan"] . "\n";
echo "IPK: " . $mahasiswa["ipk"] . "\n";
?>
